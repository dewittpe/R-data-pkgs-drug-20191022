1 #' Import PSPS Data
  2 #'
  3 #' Create \code{tbl_spark} objects for each year of the PSPS data.
  4 #'
  5 #' @param sc a sparklyr culster
  6 #' @param envir the envir to assign the \code{tbl_spark} object too, defatuls to
  7 #' the global environment.
  8 #' @param ... arguments passed to \code{\link[sparklyr]{spark_read_csv}} (for
  9 #' the import of 2017 data) or to \code{\link[sparklyr]{spark_read_text}} for
 10 #' the years 2004 through 2016.
 11 #'
 12 #' @return A \code{tbl_spark} object of the name psps_<year> will be in the
 13 #' \code{envir}.
 14 #'
 15 #' @examples
 16 #' \donttest{
 17 #'
 18 #' sc <- sparklyr::spark_connect("local")
 19 #'
 20 #' ls()
 21 #' import_psps_2004(sc, memory = FALSE)
 22 #' import_psps_2005(sc, memory = FALSE)
 23 #' import_psps_2006(sc, memory = FALSE)
 24 #' import_psps_2007(sc, memory = FALSE)
 25 #' import_psps_2008(sc, memory = FALSE)
 26 #' import_psps_2009(sc, memory = FALSE)
 27 #' import_psps_2010(sc, memory = FALSE)
 28 #' import_psps_2011(sc, memory = FALSE)
 29 #' import_psps_2012(sc, memory = FALSE)
 30 #' import_psps_2013(sc, memory = FALSE)
 31 #' import_psps_2014(sc, memory = FALSE)
 32 #' import_psps_2015(sc, memory = FALSE)
 33 #' import_psps_2016(sc, memory = FALSE)
 34 #' import_psps_2017(sc, memory = FALSE)
 35 #' ls()
 36 #'
 37 #'
 38 #' psps_data <-
 39 #'   sparklyr::sdf_bind_rows(psps_2004, psps_2005, psps_2006,
 40 #'                           psps_2007, psps_2008, psps_2009,
 41 #'                           psps_2010, psps_2011, psps_2012,
 42 #'                           psps_2013, psps_2014, psps_2015,
 43 #'                           psps_2016, psps_2017)
 44 #'
 45 #' qwraps2::frmt(sparklyr::sdf_nrow(psps_data))
 46 #'
 47 #' sparklyr::spark_disconnect(sc)
 48 #'
 49 #' }
 50 #'
 51 #' @name import
 52 NULL
 53 
 54 #' @rdname import
 55 #' @export
 56 import_psps_2004 <- function(sc, envir = .GlobalEnv, ...) {
 57   tmpdir <- tempdir()
 58   message("Unziping PSPS 2004 source data")
 59 
 60   utils::unzip(system.file("extdata", "PSPS_2004.zip", package = "psps.data"),
 61                exdir = paste(tmpdir, "psps_2004", sep = "/"))
 62 
 63   message("Building psps_2004")
 64 
 65   files <- list.files(path = paste(tmpdir, "psps_2004", sep = "/"),
 66                       full.names = TRUE,
 67                       recursive = TRUE,
 68                       pattern = "\\.exe$")
 69 
 70   for (f in files) {
 71     utils::unzip(f, exdir = paste(tmpdir, "psps_2004", "data", sep = "/"))
 72   }
 73 
 74   files <- list.files(path = paste(tmpdir, "psps_2004", "data", sep = "/"),
 75                       full.names = TRUE,
 76                       recursive = TRUE,
 77                       pattern = "B04")
 78 
 79   object <-
 80     sparklyr::sdf_bind_rows(lapply(files,
 81                                    function(x, ...) {
 82                                      sparklyr::spark_read_text(sc, name = x, path = x, ...)
 83                                    }))
 84 
 85   assign(x     = "psps_2004",
 86          value = psps_mutate_v1(object, YEAR = 2004),
 87          envir = envir)
 88 
 89   message("psps_2004 is available")
 90   invisible(TRUE)
 91 }

586                 HCPCS_CD                    =            substr(.data$line, start =   1L, stop =   5L),
587                 HCPCS_INITIAL_MODIFIER_CD   =            substr(.data$line, start =   6L, stop =   7L),

593                 HCPCS_SECOND_MODIFIER_CD    =            substr(.data$line, start =  20L, stop =  21L),
594                 REGION                      =            substr(.data$line, start =  22L, stop =  23L),
595                 PSPS_SUBMITTED_SERVICE_CNT  = as.numeric(substr(.data$line, start =  24L, stop =  32L)),
596                 MTUS                        = as.numeric(substr(.data$line, start =  33L, stop =  41L)),
597                 ANESTHESIA_UNITS            = as.numeric(substr(.data$line, start =  33L, stop =  41L)),
